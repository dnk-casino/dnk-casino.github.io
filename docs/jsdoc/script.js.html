/**
 * Funci칩n para comprobar si el token es v치lido
 * @returns {boolean} - True si el token es v치lido, false de lo contrario
 */
function checkAuth() {
    try {
        const token = JSON.parse(atob(localStorage.getItem('token').split('.')[1]));
        return token.exp > Math.floor(Date.now() / 1000);
    } catch (error) {
        return false;
    }
}
